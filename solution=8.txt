#include <iostream>
using namespace std;

enum enMark
{
	PASS=1,FAIL=2
};

int ReadMark()
{
	int Mark;
	cout << "Please Enter Your Mark: " << endl;
	cin >> Mark;
	return Mark;
}
enMark IsAccespted(int Mark)
{
	if (Mark >= 50)
		return enMark::PASS;
	else
		return enMark::FAIL;
}
void PrintMark(enMark IMark)
{
	if (IMark == enMark::PASS)
		cout << "\n Number is PASS." << endl;
	else
		cout << "\n Number is FAIL." << endl;
}
int main()
{
	PrintMark(IsAccespted(ReadMark()));
	return 0;
}
