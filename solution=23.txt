#include <iostream>
#include <string>
using namespace std;
void ReadNumbers(float& A, float& B, float& C)
{
	cout << "Please Enter Your Number A: " << endl;
	cin >> A;
	cout << "Please Enter Your Number B: " << endl;
	cin >> B;
	cout << "Please Enter Your Number C: " << endl;
	cin >> C;
}
float CircleAreaDescribedAroundAnArbitraryTriangle(float A, float B, float C)
{
	const float PI = 3.141592653589793238;
	float P = (A + B + C) / 2;
	float Area = PI * pow((A * B * C) / (4 * sqrt(P * (P - A) * (P - B) * (P - C))), 2);
	return Area;
}
void PrintResult(float Area)
{
	cout << "\n Circle Area is: " << Area << "." << endl;
}
int main()
{
	float A, B, C;
	ReadNumbers(A, B, C);
	PrintResult(CircleAreaDescribedAroundAnArbitraryTriangle(A, B, C));
	return 0;
}
