#include <iostream>
#include <string>
using namespace std;
enum enMark
{
	PASS=1,FAIL=2
};
int ReadMark()
{
	int Mark;
	cout << "Please Enter Your Mark: " << endl;
	cin >> Mark;
	return Mark;
}
enMark CheckMark(int Mark)
{
	if (Mark >= 50)
		return enMark::PASS;
	else
		return enMark::FAIL;
}
void PrintMark(int Mark)
{
	if (CheckMark(Mark) == enMark::PASS)
		cout << "\n Passed." << endl;
	else
		cout << "\n Faild." << endl;
}
int main()
{
	PrintMark(ReadMark());
	return 0;
}
