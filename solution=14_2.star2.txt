#include <iostream>
#include <string>
using namespace std;
int ReadPositiveNumber(string Message)
{
	int Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number <= 0);
	return Number;
}
void PrintNumberPattern(int Number)
{
	
	cout << "\n";
	for (int i = (Number + 65) - 1; i >= 65; i--)
	{
		for (int j = 65; j <= i; j++)
		{
			
			cout << char (j) << "";
			
		}
		cout << "\n";
	}
}
int main()
{
	PrintNumberPattern(ReadPositiveNumber("Please Enter Your Positive Number: "));
	return 0;
}
