#include <iostream>
using namespace std;
int ReadNumber()
{
	int Num = 0;
	cout << "Enter Number:\n";
	cin >> Num;
	return Num;
}
void CheckNumber(int Num)
{
	int* p;
	p = new int[Num];
	cout << "The Employees is: " << endl;
	for (int i = 0; i < Num; i++)
	{
		cout << "\nEnter Employees: ";
		cin >> *(p + i);
	}
	cout << "The Result: \n";
	for (int i = 0; i < Num; i++)
	{
		cout << "Employee " << i + 1 << ": " << *(p + i) << endl;
	}
	delete[] p;
}
int main()
{
	CheckNumber(ReadNumber());
	return 0;
}
