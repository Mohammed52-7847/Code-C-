#include <iostream>
#include <string>
using namespace std;
void ReadTriangle(float& A,float& B)
{
	cout << "Please Enter Triangle A: " << endl;
	cin >> A;
	cout << "Please Enter Triangle B: " << endl;
	cin >> B;
}
float CircleAreaInscribedInAnIsoScelesTriangle(float A, float B)
{
	const float PI = 3.141592653589793238;
	float Area = PI * (pow(B, 2) / 4) * ((2 * A) - B) / ((2 * A) + B);
	return Area;
}
void PrintResultTriangle(float Area)
{
	cout << "\n Circle Area is: " << Area << "." << endl;
}
int main()
{
	float A, B;
	ReadTriangle(A, B);
	PrintResultTriangle(CircleAreaInscribedInAnIsoScelesTriangle(A, B));
	return 0;
}
