#include <iostream>
#include <string>
using namespace std;

void ReadNumber(int& Num1, int& Num2)
{
	cout << "Please Enter Your Number 1: " << endl;
	cin >> Num1;
	cout << "Please Enter Your Number 2: " << endl;
	cin >> Num2;
}
void Swap(int& A, int& B)
{
	int Temp;
	Temp = A;
	A = B;
	B = Temp;
}
void PrintSwap(int Num1, int Num2)
{
	cout << "\n Number 1: " << Num1 << "." << endl;
	cout << "\n Number 2: " << Num2 << "." << endl;
	cout << "\n -----------------------\n";
}
int main()
{
	int Num1, Num2;
	ReadNumber(Num1,Num2);
	PrintSwap(Num1,Num2);
	Swap(Num1,Num2);
	PrintSwap(Num1,Num2);
	return 0;
}
