#include <iostream>
using namespace std;
void PrintArray()
{
	int arr[10] = { 10,10,10,50,50,70,70,70,70,90 };
	for (int i = 0; i < 10; i++)
	{
		cout << arr[i] << " ";
		
	}
	cout << "\nArray 2 Elements:\n";
	cout << arr[0] << " " << arr[3] << " " << arr[5] << " " << arr[9] << endl;
	cout << "\n";
}

int main()
{
	cout << "\nArray 1 Elements:\n";
	PrintArray();
	return 0;
}
