#include <iostream>
#include <cmath>
using namespace std;
double MyFunction(int a, int b, const float pi)
{
	return (pi * pow(b, 2) / 4 * ((2 * a) - b) / ((2 * a) + b));
}
int main()
{
	int a;
	int b;
	const float pi = 3.14;
	cout << "please enter your number-1? ";
	cin >> a;
	cout << "please enter your number-2? ";
	cin >> b;

	cout << MyFunction(a, b, pi) << endl;

	return 0;
}
