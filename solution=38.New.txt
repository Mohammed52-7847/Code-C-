#include <iostream>
#include <string>
using namespace std;
enum enPrimeAndNotPrime
{
	Prime = 1,NotPrime = 2
};
float ReadPositiveNumber(string Message)
{
	float Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number <= 0);
	return Number;
}
enPrimeAndNotPrime CheckPrime(int Number)
{
	int M = round(Number / 2);
	for (int Counter = 2; Counter <= M; Counter++)
	{
		if (Number % Counter == 0)
			return enPrimeAndNotPrime::NotPrime;
	}
	return enPrimeAndNotPrime::Prime;
}
void PrintNumberType(int Number)
{
	switch (CheckPrime(Number))
	{
	case enPrimeAndNotPrime::Prime:
		cout << "The Number is Prime\n";
		break;
	case enPrimeAndNotPrime::NotPrime:
		cout << "The Number is NotPrime\n";
		break;
	}
}
int main()
{
	PrintNumberType(ReadPositiveNumber("Please Enter a Positive Number: "));
	return 0;
}
