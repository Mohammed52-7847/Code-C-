#include <iostream>
using namespace std;
int MyCeil(float Number)
{
	int IntPart;
	IntPart = int(Number);

	if (Number > 0)
	{
		return ++IntPart;
	}
	else
	{
		return IntPart;
	}
}
float ReadNumber()
{
	float Number;
	cout << "Please Enter a Number: ";
	cin >> Number;
	return Number;
}
int main()
{
	float Number = ReadNumber();
	cout << "My My Ceil Result: " << MyCeil(Number) << endl;
	cout << "C++ Ceil Result: " << ceil(Number) << endl;
	return 0;
}
