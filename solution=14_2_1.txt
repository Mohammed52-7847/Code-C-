#include <iostream>
#include <string>
using namespace std;
int ReadPositiveNumber(string Message)
{
	int Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number <= 0);
	return Number;
}
void PrintNumberPattern(int Number)
{
	
	cout << "\n";
	char Letter = 'A';
	for (int i = Number; i >= 1; i--)
	{
		for (int j = 1; j <= i; j++)
		{
			
			cout << Letter << " ";
			Letter++;
		}
		
		cout << "\n";
	}
}
int main()
{
	PrintNumberPattern(ReadPositiveNumber("Please Enter Your Positive Number: "));
	return 0;
}
