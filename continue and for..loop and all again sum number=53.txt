#include <iostream>
using namespace std;

int main()
{
	int number = 0;
	int sum = 0;
	for (int i = 1; i <= 5; i++)
	{
		cout << "Please Enter Your Number: ";
		cin >> number;
		if (number > 50)
		{
			cout << "Please Enter Your Number Again.\n";
			continue;
		}
		sum += number;
		cout << "The SUM is: " << sum << endl;
	}
	

	return 0;
}
