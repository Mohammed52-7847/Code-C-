#include <iostream>
#include <string>
using namespace std;
string ReadPassWord()
{
	string PassWord;
	cout << "Please Enter a 3-Letter PassWord (all Capital)?\n";
	cin >> PassWord;
	return PassWord;
}
bool GuessPassWord(string OriginalPassWord)
{
	string word = "";
	int Counter = 0;
	cout << "\n";
	for (int i = 65; i <= 90; i++)
	{
		for (int j = 65; j <= 90; j++)
		{
			for (int k = 65; k <= 90; k++)
			{
				word = word + char(i);
				word = word + char(j);
				word = word + char(k);
				Counter++;
				cout << "Trial is [" << Counter << "]: ";
				cout << word << endl;
				if (word == OriginalPassWord)
				{
					
					cout << "\nPassWord is " << word << "\n";
					cout << "Found AFter ";
					cout << Counter << " Trial(s).\n";
					return true;
				}
				word = "";
			}
		}
	}
	return false;
}
int main()
{
	GuessPassWord(ReadPassWord());
	return 0;
}
