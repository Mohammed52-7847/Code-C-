#include <iostream>
#include <string>
using namespace std;
void PrintFibonacciUsingLoop(short N)
{
	short FebNumber = 0, Prev2 = 0, Prev1 = 1;
	cout << "1  ";

	for (short i = 2; i <= N; i++)
	{
		
		FebNumber = Prev1 + Prev2;
		cout << FebNumber << "   ";
		Prev2 = Prev1;
		Prev1 = FebNumber;
		
	}
}
int main()
{
	PrintFibonacciUsingLoop(10);
	cout << endl;
	system("pause>0");
	return 0;
}
