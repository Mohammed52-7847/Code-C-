#include <iostream>
#include <string>
using namespace std;
int ReadHalfNumber()
{
	int Num;
	cout << "Please Enter Your Number: " << endl;
	cin >> Num;
	return Num;
}
float CalculateHalfNum(int Num)
{
	return (float)Num / 2;
}
void PrintHalfNum(int Num)
{
	string Result = "Half of: " + to_string(Num) + " is " + to_string(CalculateHalfNum(Num));
	cout << endl << Result << endl;
}
int main()
{
	PrintHalfNum(ReadHalfNumber());
	return 0;
}
