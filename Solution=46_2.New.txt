#include <iostream>
using namespace std;
float MyABS(float Number)
{
	if (Number > 0)
	{
		return Number;
	}
	else
	{
		return Number * -1;
	}
}
float ReadNumber()
{
	float Number;
	cout << "\nPlease Enter a Number? ";
	cin >> Number;
	return Number;
}
int main()
{
	float Number = ReadNumber();
	cout << "\nMy abs Result: ";
	cout << MyABS(Number) << endl;
	cout << "\nC++ abs Result: ";
	cout << abs(Number) << endl;
	return 0;
}
