#include <iostream>
#include <string>
using namespace std;
enum enDays
{
	 Sunday=1,Monday=2,Tusday=3,Wednesday=4,Thursday=5,Friday=6,Saturday=7,WrongDay=8
};
enDays ReadNumOfDay()
{
	int Number = 0;
	do
	{
		cout << "Please Enter Your Day Again: " << endl;
		cin >> Number;
	} while (Number < 1 || Number > 7);
	return (enDays)Number;
}
string CheckDays(enDays Day)
{
	switch (Day)
	{
	case enDays::Sunday:
		return "\n It is Sunday";
		break;
	case  enDays::Monday:
		return "\n It is Monday";
		break;
	case enDays::Tusday:
		return "\n It is Tusday";
		break;
	case enDays::Wednesday:
		return "\n It is Wednesday";
		break;
	case enDays::Thursday:
		return "\n It is Thursday";
		break;
	case enDays::Friday:
		return "\n It is Friday";
		break;
	case enDays::Saturday:
		return "\n It is Saturday";
		break;
	}
}
int main()
{
	cout << CheckDays(ReadNumOfDay()) << endl;
	return 0;
}
