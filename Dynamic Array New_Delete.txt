#include <iostream>
using namespace std;
int main()
{
	int Num = 0;
	cout << "Enter Total Number of Student:\n";
	cin >> Num;
	float* ptr;
	ptr = new float[Num];
	cout << "Enter Grades Of Students.\n";
	for (int i = 0; i < Num; ++i)
	{
		cout << "Student " << i + 1 << ": ";
		cin >> *(ptr + i);
	}
	cout << "\nDisPlaying Gardes Of Students.\n";
	for (int i = 0; i < Num; ++i)
	{
		cout << "Student " << i + 1 << ": " << *(ptr + i) << endl;
	}
	delete[]ptr;

	return 0;
}
