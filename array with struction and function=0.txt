#include <iostream>
using namespace std;

struct strinfo
{
	string firstname;
	string lastname;
	int age;
	string phone;
};
void Readprocedure2(strinfo& person)
{
	cout << "please enter your firstname: ";
	cin >> person.firstname;
	cout << "please enter your lastname: ";
	cin >> person.lastname;
	cout << "please enter your age: ";
	cin >> person.age;
	cout << "please enter your phone: ";
	cin >> person.phone;
}
void printprocedure2(strinfo person)
{
	cout << "FirstName: " << person.firstname << "." << endl;
	cout << "LastName: " << person.lastname << "." << endl;
	cout << "Age: " << person.age << "." << endl;
	cout << "Phone: " << person.phone << "." << endl;
	cout << "\n*******************************\n";

}
void Readprocedure(strinfo info[2])
{
	Readprocedure2(info[0]);
	Readprocedure2(info[1]);
}
void printprocedure(strinfo info[2])
{
	printprocedure2(info[0]);
	printprocedure2(info[1]);

}


int main()
{
	strinfo info[2];
	Readprocedure(info);
	printprocedure(info);
	return 0;
}
