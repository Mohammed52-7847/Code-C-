#include <iostream>
#include <string>
using namespace std;
enum enPrimAndNotPrime{Prime = 1,NotPrime = 2};
int ReadNumber()
{
	int Number = 0;
	do
	{
		cout << "Enter Your Number:\n";
		cin >> Number;
	} while (Number < 0);
	return Number;
}
enPrimAndNotPrime CheckPrimNotPrime(int Number)
{
	int M = round(Number / 2);
	for (int Counter = 2; Counter <= M; Counter++)
	{
		if (Number % Counter == 0)
		{
			return enPrimAndNotPrime::NotPrime;
		}
		else
		{
			return enPrimAndNotPrime::Prime;
		}
	}
}
void PrintResult(int Num)
{
	if (CheckPrimNotPrime(Num)==enPrimAndNotPrime::NotPrime)
	{
		cout << "\nNot Prime.\n";
	}
	else
	{
		cout << "\nPrime.\n";
	}
}
int main()
{
	PrintResult(ReadNumber());
	return 0;
}
