enWeekDay CheckDay()
{
	enWeekDay Day;
	int c = 0;
	cin >> c;

}
string PrintDay(enWeekDay Day)
{
		switch (Day)
		{
		case enWeekDay::Saturday:
			system("Color 1F");
			return "Saturday";
			break;
		case enWeekDay::Sunday:
			system("Color 2F");
			return "Sunday";
			break;
		case enWeekDay::Monday:
			system("Color 3F");
			return "Monday";
			break;
		case enWeekDay::Tuesday:
			system("Color 5F");
			return "Tuesday";
			break;
		case enWeekDay::Wednesday:
			system("Color 6F");
			return "Wednesday";
			break;
		case enWeekDay::Thursday:
			system("Color 7F");
			return "Thursday";
			break;
		case enWeekDay::Friday:
			system("Color 8F");
			return "Friday";
			break;
		default:
			system("Color 4F");
			return "Wrong Day";
			break;
		}
		
}
