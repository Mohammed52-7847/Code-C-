#include <iostream>
#include <string>
using namespace std;
int ReadPositiveNumber(string Message)
{
	int Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number < 0);
	return Number;
}
int CountPalinderNumber(int Number)
{
	int Remainder = 0,Number2 = 0;
	while (Number > 0)
	{
		Remainder = Number % 10;
		Number = Number / 10;
		Number2 = Number2 * 10 + Remainder;
	}
	return Number2;
}
bool ReverseNumber(int Number)
{
	return Number == CountPalinderNumber(Number);
}
void PalinderResult(int Number)
{
	if (ReverseNumber(Number))
	{
		cout << "\nIt is Palinder.\n";
	}
	else
	{
		cout << "\nIt is Not Palinder.\n";
	}
}
int main()
{
	int Number = ReadPositiveNumber("Enter Palinder Number:\n");
	PalinderResult(Number);
	return 0;
}
