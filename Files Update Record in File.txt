#include <iostream>
#include <string>
#include <fstream>
#include <vector>
using namespace std;
void Load(string FileName, vector <string>& vLetter)
{
	fstream MyFile;
	MyFile.open("MyFile.txt", ios::in);
	if (MyFile.is_open())
	{
		string Line;
		while (getline(MyFile, Line))
		{
			vLetter.push_back(Line);
		}
		MyFile.close();
	}
}
void Save(string FileName, vector <string> vLetter)
{
	fstream MyFile;
	MyFile.open("MyFile.txt", ios::out);
	if (MyFile.is_open())
	{
		for (string Line : vLetter)
		{
			if (Line != "")
			{
				MyFile << Line << endl;
			}
		}
		MyFile.close();
	}
}
void Delete(string FileName, string Record,string Update)
{
	vector <string> vLetter;
	Load(FileName, vLetter);
	for (string& Line : vLetter)
	{
		if (Line == Record)
		{
			Line = Update;
		}
	}
	Save(FileName, vLetter);
}
