#include <iostream>
#include <string>
using namespace std;
int ReadNumber()
{
	int Number;
	cout << "Please Enter Your Number: " << endl;
	cin >> Number;
	return Number;
}
int ReadPower()
{
	int Number;
	cout << "Please Enter Your Power: " << endl;
	cin >> Number;
	return Number;
}
int PowerNumber(int Number, int M)
{
	if (M == 0)
	{
		return 1;
	}
	int p = 1;
	for (int i = 1; i <= M; i++)
	{
		p *= Number;
	}
	return p;
}
int main()
{
	cout << "Result: " << PowerNumber(ReadNumber(), ReadPower());
	return 0;
}
