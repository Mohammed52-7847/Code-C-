#include <iostream>
using namespace std;
void CheckNumber(int arr[100],int &Length)
{
	cout << "Enter Number:\n";
	cin >> Length;
	for (int i = 0; i < Length; i++)
	{
		cout << "Enter Array " << i+1 << ": " << endl;
		cin >> arr[i];
	}
}
void PrintArray(int arr[100], int Length)
{
	for (int i = 0; i < Length; i++)
	{
		cout << arr[i] << endl;
	}
}
int main()
{
	int arr[100],Length;
	CheckNumber(arr, Length);
	PrintArray(arr, Length);
	return 0;
}
