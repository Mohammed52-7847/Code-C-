#include <iostream>
using namespace std;
int ReadNumber()
{
	int Length;
	cout << "Enter Number:\n";
	cin >> Length;
	return Length;
}
void CheckNumber(int arr[100],int Length)
{
	for (int i = 0; i < Length; i++)
	{
		cout << "Enter Array " << i+1 << ": " << endl;
		cin >> arr[i];
	}
}
void PrintArray(int arr[100], int Length)
{
	for (int i = 0; i < Length; i++)
	{
		cout << arr[i] << endl;
	}
}
int main()
{
	int arr[100];
	int Length = ReadNumber();
	CheckNumber(arr, Length);
	PrintArray(arr, Length);
	return 0;
}
