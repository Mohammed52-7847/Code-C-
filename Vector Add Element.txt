#include <vector>
#include <iostream>
using namespace std;
int main()
{
	vector <float> vNumber;
	vNumber.push_back(12.53);
	vNumber.push_back(5.2);
	vNumber.push_back(7.56);
	vNumber.push_back(60.25);
	vNumber.push_back(90.5);
	cout << "The Vectors Numbers is: ";
	for (float Number : vNumber)
	{
		cout << Number << " ";
	}
	cout << endl;

	return 0;
}
