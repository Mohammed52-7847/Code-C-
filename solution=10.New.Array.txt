#include <iostream>
#include <string>
using namespace std;
void ReadMark(int Mark[3])
{
	cout << "PLease Enter Your Mark 1: " << endl;
	cin >> Mark[0];
	cout << "Please Enter Your Mark 2: " << endl;
	cin >> Mark[1];
	cout << "Please Enter Your Mark 3: " << endl;
	cin >> Mark[2];
}
int SumOf3Num(int Mark[3])
{
	return Mark[0] + Mark[1] + Mark[2];
}
float Average(int Mark[3])
{
	return (float)SumOf3Num(Mark) / 3;
}
void PrintAverage(float Average)
{
	cout << "\n The Average is: " << Average << "." << endl;
}
int main()
{
	int Mark[3];
	ReadMark(Mark);
	PrintAverage(Average(Mark));
	return 0;
}
