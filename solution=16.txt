#include <iostream>
#include <cmath>
#include <string>
using namespace std;
void ReadNumbers(int& a, int& d)
{
	cout << "Please Enter Your Number a: " << endl;
	cin >> a;
	cout << "Please Enter Your Number d: " << endl;
	cin >> d;
}
float CalculateArea(float a, float d)
{
	return a * sqrt(pow(d, 2) - pow(a, 2));
}
void PrintResults(float Area)
{
	cout << "The Result is: " << Area << "." << endl;
}
int main()
{
	int a, d;
	ReadNumbers(a, d);
	PrintResults(CalculateArea(a, d));
	return 0;
}
