#include <iostream>
#include <string>
using namespace std;
int main()
{
	for (int i = 65; i <= 90; i++)
	{
		cout << "Letter is: " << char(i) << endl;
		for (int j = 65; j <= 90; j++)
		{
			for (int k = 65; k <= 90; k++)
			{
				cout << char(i) << char(j) << char(k) << "\n";
			}
		}
		cout << "\n_____________\n";
	}
	return 0;
}
