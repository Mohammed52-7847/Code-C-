#include <iostream>
#include <string>
using namespace std;
float ReadSquareSide()
{
	float A;
	cout << "Please Enter Square Side A: " << endl;
	cin >> A;
	return A;
}
float CircleAreaSquareSide(float A)
{
	const float PI = 3.141592653589793238;
	float Area = PI * (pow(A / 2, 2));
	return Area;
}
void PrintResultSquare(float Area)
{
	cout << "\n Circle Area Square: " << Area << "." << endl;
}
int main()
{
	PrintResultSquare(CircleAreaSquareSide(ReadSquareSide()));
	return 0;
}
