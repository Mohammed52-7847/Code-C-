#include <iostream>
#include <cstdlib>
#include <iomanip>
using namespace std;
int RandomNumber(int From, int To)
{
	int randNum = rand() % (To - From + 1) + From;
	return randNum;
}
void FillArray(int arr[3][3])
{
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			arr[i][j] = RandomNumber(1, 10)*RandomNumber(1,10);
		}
		cout << endl;
	}
	
}
void PrintArray(int arr[3][3])
{
	for (int i = 0; i < 3; i++)
	{
		
		for (int j = 0; j < 3; j++)
		{
			cout << setw(8) << arr[i][j] << " ";
		}
		cout << endl;
	}

}
int main()
{
	srand((unsigned)time(NULL));
	int arr[3][3];
	FillArray(arr);
	cout << setw(35) << "______________________________________\n\n";
	cout << setw(35) << "The Following is a 3*3 Random Matrix: " << endl;
	cout << setw(35) << "______________________________________\n\n";
	PrintArray(arr);
	return 0;
}
