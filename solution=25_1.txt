#include <iostream>
#include <string>
using namespace std;
int ReadNumber()
{
	int Age = 0;
		cout << "Enter Your Age:\n";
		cin >> Age;
	return Age;
}
bool CalculateNumber(int Age)
{
	return (Age >= 18 && Age <= 45);
}
int CheckAge(int Age)
{
	do
	{
		cout << "Enter Your Age:\n";
		cin >> Age;
	} while (!CalculateNumber(Age));
	return Age;
}
void PrintResult(int Age)
{
	if (CalculateNumber(Age))
	{
		cout << "\nValid.\n";
	}
	else if (!CalculateNumber(Age))
	{
		cout << "\nInvalid.\n";
	}
}
int main()
{
	PrintResult(CheckAge(CalculateNumber(ReadNumber())));
	return 0;
}
