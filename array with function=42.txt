#include <iostream>
#include <cmath>
using namespace std;

void Readprocedure(double x[4])
{
	cout << "please enter number of day: ";
	cin >> x[0];
	cout << "please enter number of hours: ";
	cin >> x[1];
	cout << "please enter number of minutes: ";
	cin >> x[2];
	cout << "please enter number of seconds: ";
	cin >> x[3];
}
double secondsfunction(double x[4])
{
	return x[0] * (24 * 60 * 60) + x[1] * (60 * 60) + x[2] * (60) + x[3];
}
void printprocedure(double x[4])
{
	cout << "\n**********************************\n";
	cout << "Total Seconds: " << secondsfunction(x) << "." << endl;
	cout << "\n**********************************\n";
}
int main()
{
	double x[4];
	Readprocedure(x);
	secondsfunction(x);
	printprocedure(x);


	return 0;
}
