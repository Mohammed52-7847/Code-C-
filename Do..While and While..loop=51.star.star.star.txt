#include <iostream>
using namespace std;
int ReadIntNumberInRangeUsingDoWhile(int From, int To)
{
	int Number;
	do
	{
		cout << "Please Enter a Number between " << From << " and " << To << endl;
		cin >> Number;
	} while (Number < From || Number > To);
	return Number;
}
