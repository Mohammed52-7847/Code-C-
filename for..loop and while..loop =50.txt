#include <iostream>
using namespace std;

int main()
{
	cout << "for...loop: " << endl;
	for (int i = 1; i <= 5; i++)
	{
		cout << i << " " << endl;
	}
	cout << "-------------------------\n";
	cout << "While....loop: " << endl;
	int i = 1;
	while (i <= 5)
	{
		cout << i <<  " ";
		i++;
	}

	return 0;
}
