#include <iostream>
using namespace std;

int main()
{
	int number1;
	int number2;
	char type;
	cout << "please enter your number1: ";
	cin >> number1;
	cout << "please enter your number2: ";
	cin >> number2;
	cout << "please enter your type: ";
	cin >> type;

	switch (type)
	{
	case('+'):
		cout << number1 + number2 << "." << endl;
		break;
	case('-'):
		cout << number1 - number2 << "." << endl;
		break;
	case('*'):
		cout << number1 * number2 << "." << endl;
		break;
	case('/'):
		cout << number1 / number2 << "." << endl;
		break;
	default:
		cout << "Other Type." << endl;
	}
	return 0;
}
