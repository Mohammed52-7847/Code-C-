#include <iostream>
using namespace std;
enum enScreenColor
{
	Red=1,Pink=2,Green=3,White=4,Orange=5,Blue=6
};
int main()
{
	cout << "***************************************\n";
	cout << "please chose the number of your color?\n";
	cout << "(1) Red\n";
	cout << "(2) Pink\n";
	cout << "(3) Green\n";
	cout << "(4) White\n";
	cout << "(5) Orange\n";
	cout << "(6) Blue\n";
	cout << "***************************************\n\n";
	cout << "Your Choice? ";

	int c;
	enScreenColor Color;
	cin >> c;
	Color = (enScreenColor)c;
	if (Color == enScreenColor::Red)
	{
		system("color 1F");
	}
	else if (Color == enScreenColor::Pink)
	{
		system("color 2F");
	}
	else if (Color == enScreenColor::Green)
	{
		system("color 3F");
	}
	else if (Color == enScreenColor::White)
	{
		system("color 4F");
	}
	else if (Color == enScreenColor::Orange)
	{
		system("color 5F");
	}
	else if (Color == enScreenColor::Blue)
	{
		system("color 6F");
	}
	else
	{
		system("color 7F");
	}

	return 0;
}
