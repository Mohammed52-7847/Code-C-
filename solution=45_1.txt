#include <iostream>
#include <string>
using namespace std;
enum enDay{Sunday = 1,Monday = 2,Tuesday = 3,Wednesday = 4,Thursday = 5,Friday = 6,Saturday = 7};
int ReadNumber()
{
	int Number = 0;
	do
	{
		cout << "Enter Your Day:\n";
		cin >> Number;
	} while (Number < 1 || Number > 7);
	return Number;
}
enDay GetDay()
{
	return (enDay)ReadNumber();
}
