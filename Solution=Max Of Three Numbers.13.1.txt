};

int CheckNumber(strInfo Info)
{
	if (Info.A > Info.B)
	{
		if (Info.A > Info.C)
		{
			return Info.A;
		}
		else
		{
			return Info.C;
		}
	}
	else if (Info.B > Info.C)
	{
		return Info.B;
	}
	else
	{
		return Info.C;
	}
	
}
