#include <iostream>
using namespace std;
int MySqrt(float Number)
{
	return pow(Number, 0.5);
}
float ReadNumber()
{
	float Number;
	cout << "Please Enter a Number?\n";
	cin >> Number;
	return Number;
}
int main()
{
	float Number = ReadNumber();
	cout << "My MySqrt Result: " << MySqrt(Number) << endl;
	cout << "C++ Sqrt Result: " << sqrt(Number) << endl;

	return 0;
}
