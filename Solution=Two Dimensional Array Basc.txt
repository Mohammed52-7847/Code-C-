#include <iostream>
#include <iomanip>
using namespace std;
void Fill(int arr[5][5], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			cout << "Enter " << " Number: ";
			cin >> arr[i][j];
		}
		cout << endl;
	}
}
void PrintMatrix(int arr[5][5], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			cout << setw(3) << arr[i][j] << "   ";
		}
		cout << endl;
	}
}
int main()
{
	int arr[5][5];
	Fill(arr, 5, 5);
	PrintMatrix(arr, 5, 5);
	
	system("pause > 0");
}
