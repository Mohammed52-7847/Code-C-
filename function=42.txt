#include <iostream>
#include <cmath>
using namespace std;

double Myfunction(int d, int h, int m, int s)
{
	return (d * 24 * 60 * 60) + (h * 60 * 60) + (m * 60) + (s);
}
int main()
{
	int d;
	int h;
	int m;
	int s;
	cout << "please enter your day: ";
	cin >> d;
	cout << "please enter your hours: ";
	cin >> h;
	cout << "please enter your minutes: ";
	cin >> m;
	cout << "please enter your second: ";
	cin >> s;

	cout << Myfunction(d, h, m, s) << endl;

	return 0;
}
