#include <iostream>
using namespace std;
int RandomNumber(int From, int To)
{
	int randNum = rand() % (To - From + 1) + From;
	return randNum;
}
void FillArrayRandomNumbers(int arr[100], int& arrlenght)
{
	cout << "\nEnter Number Of Elements:\n";
	cin >> arrlenght;
	for (int i = 0; i < arrlenght; i++)
	{
		arr[i] = RandomNumber(1, 100);
	}
}
void PrintArray(int arr[100], int arrlength)
{
	for (int i = 0; i < arrlength; i++)
	{
		cout << arr[i] << " ";
	}
	cout << "\n";
}
void AddArray(int Number, int arr[100], int& arrlength)
{
	arrlength++;
	arr[arrlength - 1] = Number;
}

void CopyArrayElements(int arrSource[100], int arrDestination[100], int arrlength, int& arrDestinationlength)
{
	for (int i = 0; i < arrlength; i++)
	{
		if (arrSource[i] % 2 == 0)
		{
			AddArray(arrSource[i], arrDestination, arrDestinationlength);
		}
	}
}
int main()
{
	srand((unsigned)time(NULL));
	int arr[100], arrlength = 0, arr2length = 0;
	FillArrayRandomNumbers(arr, arrlength);
	int arr2[100];
	CopyArrayElements(arr, arr2, arrlength, arr2length);
	cout << "\nArray 1 Elements:\n";
	PrintArray(arr, arrlength);
	cout << "\nArray 2 Elements:\n";
	PrintArray(arr2, arr2length);
	return 0;
}
