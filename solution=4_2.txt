#include <iostream>
#include <string>
using namespace std;
int ReadPositiveNumber(string Message)
{
	int Number = 0;
	do
	{
		cout << Message << endl;
		cin >> Number;
	} while (Number <= 0);
	return Number;
}
bool isPerfect(int Number)
{
	cout << "\n";
	int Sum = 0;
	for (int i = 1; i < Number; i++)
	{
		if (Number % i == 0)
			cout << i << "\t";
			Sum += i;
	}
	return Sum;
}
int main()
{
	isPerfect(ReadPositiveNumber("Please Enter Your Positive Number: "));
	cout << endl;
	return 0;
}
