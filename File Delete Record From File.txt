#include <iostream>
#include <string>
#include <fstream>
#include <vector>
using namespace std;
void Load(string FileName, vector <string>& vSaveLetter)
{
	fstream MyFile;
	MyFile.open("MyFile.txt", ios::in);
	if (MyFile.is_open())
	{
		string Line;
		while (getline(MyFile, Line))
		{
			vSaveLetter.push_back(Line);
		}
		MyFile.close();
	}
}
void Save(string FileName, vector <string> vSaveLetter)
{
	fstream MyFile;
	MyFile.open("MyFile.txt", ios::out);
	if (MyFile.is_open())
	{
		for (string& Line : vSaveLetter)
		{
			if (Line != "")
			{
				MyFile << Line << endl;
			}
		}
		MyFile.close();
	}
}
void Delete(string FileName, string Record)
{

	for (string& Line : vSaveLetter)

}
