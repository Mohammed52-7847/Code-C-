#include <iostream>
#include <cmath>
#include <string>
using namespace std;
void ReadNumber(int& A)
{
	cout << "Please Enter Your Number A: " << endl;
	cin >> A;
}
float CheckCircle(float A)
{
	const float pi = 3.14;
	return (pi * pow(A, 2)) / 4;
}
void PrintResults(float Square)
{
	cout << "The Result is: " << Square << "." << endl;
}
int main()
{
	int A;
	ReadNumber(A);
	PrintResults(CheckCircle(A));
	return 0;
}
